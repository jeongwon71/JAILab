**The kernels to be used**

(1) Squared Exponential Kernel (SE) \begin{equation} k(x,y)=\sigma^2 \textrm{exp}\left(-\frac{(x-y)^2}{2l^2}\right) \end{equation}

(2) Matern 5/2 Kernel (MT52) \begin{equation} k(x,y)=\sigma^2 \left( 1+\frac{\sqrt5|x-y|}{l} + \frac{5(x-y)^2}{3l^2} \right) \textrm{exp}\left(-\frac{\sqrt5 |x-y|}{l}\right) \end{equation}

(3) Matern 3/2 Kernel (MT32) \begin{equation} k(x,y)=\sigma^2 \left( 1+\frac{\sqrt3|x-y|}{l}\right) \textrm{exp}\left(-\frac{\sqrt3 |x-y|}{l}\right) \end{equation}

(4) Exponential Kernel (EP) \begin{equation} k(x,y)=\sigma^2 \textrm{exp}\left(-\frac{|x-y|}{l}\right) \end{equation}

(5) Rational Quadratic Kernel (RQ) \begin{equation} k(x,y)=\sigma^2 \left(1+\frac{(x-y)^2}{2\alpha l^2} \right)^{-\alpha} \end{equation}

(6) Periodic Kernel (Per) \begin{equation} k(x,y)=\sigma^2 \textrm{exp}\left(-\frac{2\textrm{sin}^2(\pi|x-y|p)}{l^2}\right) \end{equation}

# Define a variety of the kernel funcitons
                            
                            def GP_kernels(x,y,ker="SE",sigma=1, length=0.1, ap=1):
                                xy = x-y
                                xy2 = np.dot(xy.T,xy)
                                abs_xy = np.sqrt(xy2)
                                
                                # Squared Exponential Kernel
                                if ker == "SE": 
                                    return pow(sigma,2)*np.exp(-xy2/(2*pow(length,2)))
                                # Matern 5/2 Kernel
                                elif ker == "MT52": 
                                    return pow(sigma,2)*np.exp(-np.sqrt(5)*abs_xy/length)\
                                *(1+np.sqrt(5)*abs_xy/length+5*xy2/(3*pow(length,2)))
                                # Matern 3/2 Kernel
                                elif ker == "MT32":
                                    return pow(sigma,2)*np.exp(-np.sqrt(3)*abs_xy/length)*(1+np.sqrt(3)*abs_xy/length)
                                # Exponential Kernel
                                elif ker == "EP":
                                    return pow(sigma,2)*np.exp(-abs_xy/length)
                                # Rational Quadratic Kernel
                                elif ker == "RQ":
                                    return pow(sigma,2)*pow(1+xy2/(pow(length,2)*2*ap),-ap)
                                # Periodic Kernel
                                elif ker == "Per":
                                    inside = abs_xy*np.pi*ap
                                    return pow(sigma,2)*np.exp(-2*pow(np.sin(inside),2)/pow(length,2))
                                else:
                                    print("Choose a right kernel, ex) SE, MT52, MT32, EP, RQ, Per")

                            yexp = np.loadtxt("C:/Users/seo71/OneDrive - purdue.edu/Matlab_prog/PCM_Work/keff_LCT008_exp_G56_smpl")
                            Ckk = np.loadtxt("C:/Users/seo71/OneDrive - purdue.edu/Matlab_prog/Jan_Analaysis/covL.txt")
                            
                            yapp_ref = yapp[0]
                            yexp_ref = yexp[0]
                            
                            yexp_std = yexp[1:,:]-yexp_ref
                            yapp_std = yapp[1:]-yapp_ref
                            
                            X = np.zeros((250,18))
                            X[:,:17] = yexp_std[:250,]
                            X[:,17] = yapp_std[:250]
                            
                            # Calculate reference bias
                            cal = np.zeros((18,1))
                            cal[:17,0] = yexp_ref
                            cal[17,0] = yapp_ref
                            meas = np.ones((18,1))
                            d = cal-meas
                            d[17,0] = 0 
                            
                            Cmm = np.eye(18)*pow(0.0015,2)
                            Cmm[17,17] = 10**5
                            
                            dk_GLLS = -np.dot(np.dot(Ckk[17,:],np.linalg.inv(Ckk+Cmm)),d)
                            var_GLLS = Ckk[17,17] - np.dot(np.dot(Ckk[17,:], np.linalg.inv(Ckk+Cmm)), Ckk[:,17])
                            unc_GLLS = np.sqrt(var_GLLS)
                            
                            print("The reference bias with GLLS methodology is: \t\t{:.2f} pcm".format(float(dk_GLLS*10**5)))
                            print("The reference bias uncertainty with GLLS methodology is: {:.2f} pcm".format(float(unc_GLLS*10**5)))
                            
                            # Prepare for one of the hyperparameters, sigma
                            S = np.eye(18)
                            for i in range(18):
                                S[i,i] = np.sqrt(Ckk[i,i])

kernels = ["SE", "MT52", "MT32", "EP", "RQ", "Per"]
                            lvlm = np.zeros((6,2))
                            for n in range(len(kernels)):
                                lenList = np.linspace(-3,1,100)
                                ll = pow(10, lenList)
                                COV = np.zeros((18,18))
                            
                                RES = np.zeros((len(ll),2))
                                for k in range(len(ll)):
                                    for i in range(18):
                                        for j in range(18):
                                            COV[i,j] = GP_kernels(X[:,i],X[:,j],ker=kernels[n],sigma=1,length=ll[k])
                                    GP_Ckk = np.dot(np.dot(S,COV), S)
                                    GP_dk = -np.dot(np.dot(GP_Ckk[17,:],np.linalg.inv(GP_Ckk+Cmm)),d)
                                    GP_var = GP_Ckk[17,17] - np.dot(np.dot(GP_Ckk[17,:], np.linalg.inv(GP_Ckk+Cmm)), GP_Ckk[:,17])
                                    GP_unc = np.sqrt(GP_var)
                                    RES[k,0] = GP_dk*10**5
                                    RES[k,1] = GP_unc*10**5
                            
                                mv = np.min(RES[:,0])
                                lv = ll[list(RES[:,0]).index(mv)]
                                lvlm[n,0] = lv; lvlm[n,1] = mv
                                
                                plt.semilogx(ll,RES[:,0])
                                plt.scatter(lv,mv,s=15, c='r')
                                plt.xlabel('Length scale')
                                plt.ylabel('Resulting Bias (pcm)')
                                plt.title('({}) Bias variation with different lengthscale parameter with '.format(n+1)+kernels[n]+' kernel')
                                plt.show()
                                print("When the lengthscale is {0:.4f}, \n the resulting bias reaches at its minimum value, {1:.2f} pcm".format(lv,mv))
